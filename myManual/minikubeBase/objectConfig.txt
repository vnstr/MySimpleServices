apiVersion: apps/v1 # for version 1.9.0 need to use apps/v1beta2
kind: Deployment
metadata:
  name: nginx-deployment
spec:
  selector:
    matchLabels:
      app: nginx
  replicas: 2 # launch 2 pods, wich will create by template.
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:1.14.2
        ports:
        - containerPort: 80

# Required fields:
#     apiVersion - define the API-version;
#     kind - object type;
#     metadata - for identify object (name, UID and namespace);
#     spec - necessary object status.
