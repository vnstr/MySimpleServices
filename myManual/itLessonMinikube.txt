

minikube start

kubectl cluster-info

kubectl create deployment <name> --image=gcr.io/google-samples/kubernetes-bootcamp:v1

kubectl get deployments

kubectl get pods

kubectl describe pods

export POD_NAME=$(kubectl get pods -o go-template --template '{{range .items}}{{.metadata.name}}{{"\n"}}{{end}}')
echo Name of the Pod: $POD_NAME

kubectl exec $POD_NAME <command>

kubectl exec -ti $POD_NAME bash

kubectl logs <container_name>

kubectl expose deployment/kubernetes-bootcamp --type="NodePort" --port 8080

kubectl get services

kubectl label pod $POD_NAME <label_name>=<value>

kubectl get pods -l <label_name>=<value>

kubectl get rs
#rs - replicaSet

kubectl scale deployments/<name> --replicas=<number>
